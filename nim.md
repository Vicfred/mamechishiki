operator overloading
```nim
proc `*`(s: string, n: int): string =
  result = ""
  for _ in 1..n:
    result &= s

echo "a" * 5 # It works

echo 5 * 5 # The default implementation stills working
echo 45 * 89
```

operator overloading
```nim
type Restaurant = object
  name: string
  score: int
  id: int;

proc cmp(x, y: Restaurant): int =
  if x.name == y.name:
    return -cmp(x.score, y.score)
  return cmp(x.name, y.name)
```
